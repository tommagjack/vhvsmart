    async function callAPI(action, data) {

      if(!GAS_API_URL || GAS_API_URL.includes('XXXXXXXXXXXX')) { throw new Error("API URL not set"); }

      try {

        const response = await fetch(GAS_API_URL.trim(), { 

            method: 'POST', 

            body: JSON.stringify({ action, data }),

            redirect: 'follow'

        });

        const responseText = await response.text();

        let result;

        try { 

            result = JSON.parse(responseText); 

        } catch (e) { 

            console.error("GAS HTML Response snippet:", responseText.substring(0, 500));

        }

        if (!result.success) throw new Error(result.error);

        return result.data;

      } catch (error) { 

        throw error; 

      }

    }

    async function handleFileSelect(event) {

      const files = event.target.files;

      if (!files || files.length === 0) return;

      

      for (let i = 0; i < files.length; i++) {

        const file = files[i];

          const newFileObj = { id: fileId, name: file.name, mimeType: file.type, base64: base64 };

          selectedFiles.push(newFileObj);

          renderFileUI(newFileObj);

        } catch (err) { console.error("Error reading file:", err); }

      }

      

      document.getElementById('fileUpload').value = ''; 

      updateEmptyState();

    }

    const toBase64 = file => new Promise((resolve, reject) => {

      const reader = new FileReader();

      reader.readAsDataURL(file);

      reader.onload = () => resolve(reader.result);

      reader.onerror = error => reject(error);

    });

    async function handleCertFileSelect(event) {

        const file = event.target.files[0];

        if (!file) return;

        if (file.size > 5 * 1024 * 1024) { 

            showToast('ไฟล์ใหญ่เกินไป', `ไฟล์มีขนาดเกิน 5MB`, 'error'); 

            event.target.value = '';

            return; 

        }

        try {

            const base64 = await toBase64(file);

            selectedCertFile = { name: file.name, mimeType: file.type, base64: base64 };

            document.getElementById('certFileLabel').innerText = `เลือกไฟล์แล้ว: ${file.name}`;

            document.getElementById('certFileLabel').classList.replace('text-gray-600', 'text-green-600');

            document.getElementById('certFileLabel').classList.replace('text-sky-600', 'text-green-600');

        } catch (err) { console.error("Error reading file:", err); }

    }


    async function saveCertificate() {

       const certData = {

           cert_id: currentEditCertId,

           cert_name: document.getElementById('certName').value.trim(), 

           completion_date: document.getElementById('certDate').value,

       };


       if (!certData.cert_name || !certData.completion_date) { 

           showToast('ข้อมูลไม่ครบ', 'กรุณากรอก ชื่อเกียรติบัตร และ วันที่สำเร็จ', 'error'); return; 

       }

       if (!currentEditCertId && !selectedCertFile) {

           showToast('ข้อมูลไม่ครบ', 'กรุณาแนบไฟล์รูปภาพ หรือ PDF', 'error'); return; 

       }


       showLoader(currentEditCertId ? 'กำลังอัปเดตข้อมูล...' : 'กำลังอัปโหลดไฟล์และบันทึกข้อมูล...');

       try {

           const payload = { user: currentUser, data: certData, file: selectedCertFile };

           await callAPI('saveCertificate', payload); 

           showToast('สำเร็จ', 'บันทึกข้อมูลเกียรติบัตรเรียบร้อยแล้ว');

           closeModal('certModal'); 

           cachedCertificates = null; 

           loadCertificates(true);

       } catch (err) { 

           // ดักจับ Error "Load failed" ของ Safari/Mobile ที่มักเกิดหลังจากการอัปโหลดไฟล์ขนาดใหญ่เสร็จสิ้น

           const errorMsg = err.message.toLowerCase();

           if (errorMsg.includes('load failed') || errorMsg.includes('fetch') || errorMsg.includes('network')) {

               showToast('สำเร็จ', 'บันทึกข้อมูลเรียบร้อยแล้ว');

               closeModal('certModal'); 

               cachedCertificates = null; 

               loadCertificates(true);

           } else {

               showToast('ผิดพลาด', err.message, 'error'); 

           }

       } finally { 

           hideLoader(); 

       }

    }


    async function deleteCertificate(id) {

        if(!confirm('คุณแน่ใจหรือไม่ว่าต้องการลบเกียรติบัตรนี้? (ไฟล์ที่อัปโหลดไว้จะถูกนำออกจากระบบ)')) return;

        showLoader('กำลังลบข้อมูล...');

        try {

            await callAPI('deleteCertificate', { cert_id: id });

            showToast('สำเร็จ', 'ลบข้อมูลเรียบร้อยแล้ว');

            cachedCertificates = null; 

            loadCertificates(true);

        } catch(err) {

            showToast('ผิดพลาด', err.message, 'error');

        } finally { hideLoader(); }

    }


    function viewCertificate(id) {

        const cert = cachedCertificates.find(c => c.cert_id === id);

        if(!cert) return;


        const cDate = new Date(cert.completion_date).toLocaleDateString('th-TH', { year: 'numeric', month: 'long', day: 'numeric' });

        document.getElementById('viewCertName').innerText = cert.cert_name;

        document.getElementById('viewCertDate').innerHTML = `<i class="ph-fill ph-calendar-blank"></i> สำเร็จเมื่อ: ${cDate}`;

        document.getElementById('viewCertDownloadBtn').href = cert.file_url;

        

        const container = document.getElementById('certPreviewContainer');

        container.innerHTML = '<i class="ph ph-spinner animate-spin text-4xl text-gray-400"></i>';


        const previewUrl = cert.file_url.replace('/view?usp=drivesdk', '/preview').replace('/view', '/preview');


        if (cert.file_type.includes('pdf')) {

            container.innerHTML = `<iframe src="${previewUrl}" class="w-full h-[70vh] border-0" allow="autoplay"></iframe>`;

        } else {

            container.innerHTML = `<img src="${cert.file_url}" class="max-w-full max-h-[70vh] object-contain shadow-md rounded-lg">`;

            img.onerror = function() {

                container.innerHTML = `<iframe src="${previewUrl}" class="w-full h-[70vh] border-0" allow="autoplay"></iframe>`;

            };

        }


        document.getElementById('certViewModal').classList.remove('hidden');

    }
